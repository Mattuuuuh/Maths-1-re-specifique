\begin{theorem*}{Propriétés des puissances}
	Soient $a, b, c \in \Z$. On a les relations suivantes.
		\begin{gather*}
			a^{b+c} = a^{b} \times a^{c} \\
			\left(a^b\right)^c = a^{b \times c} \\
			a^{c} \times b^c = \left( a \times b \right)^c
		\end{gather*}
	En particulier, si $a \neq 0$ et pour tout $n \in \N$, on a
		\begin{align*}
			a^0 = 1, &
			&a^1 = a, &
			&a^{-n} = \dfrac1{a^n}. 
		\end{align*}
\end{theorem*}
